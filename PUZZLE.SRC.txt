****************************************************************

* PlayWithIt
*
* This contains the main event loop.
*
****************************************************************
PlayWithIt     START
               using GlobalData

Again          lda QuitFlag
               beq begin
               jmp AllDone

begin          lda Flashing
               beq skip

               lda ShowingPuzzle
               bne skip

               inc delay
               lda delay
               cmp #20
               bcc skip

               jsl Flash

               lda #$00
               sta delay

skip           lda isInterrupt
               beq noTimeUpdate

               PushLong #0
               _GetPort

               lda GlobalPort+2
               pha
               lda GlobalPort
               pha
               _SetPort

               PushWord #262
               PushWord #10
               _MoveTo

               lda Player
               cmp #$02
               beq ShowTime2

               PushLong #CurrTime
               _DrawString

               bra TimeShowed

ShowTime2      PushLong #CurrTime2
               _DrawString

TimeShowed     _SetPort

               lda #$00
               sta isInterrupt

noTimeUpdate   PushWord #0
               PushWord #$FFFF
               PushLong #EventRecord
               _TaskMaster

               pla
               bne trtevent
               jmp Again

trtevent       asl a
               tax

               jsr (TaskTable,x)

               jmp Again

AllDone        rts

TaskTable      anop
;
; Event manager events
;
               dc i'ignore'             ; 0 null
               dc i'ignore'             ; 1 mouse down
               dc i'ignore'             ; 2 mouse up
               dc i'KeyDown'            ; 3 key down
               dc i'ignore'             ; 4 undefined
               dc i'ignore'             ; 5 auto-key down
               dc i'DoUpdate'           ; 6 update event
               dc i'ignore'             ; 7 undefined
               dc i'DoActivate'         ; 8 activate
               dc i'ignore'             ; 9 switch
               dc i'ignore'             ; 10 desk acc
               dc i'ignore'             ; 11 device driver
               dc i'ignore'             ; 12 ap
               dc i'ignore'             ; 13 ap
               dc i'ignore'             ; 14 ap
               dc i'ignore'             ; 15 ap
               dc i'Ignore'             ; 0 in desk

; Task master events

               dc i'DoMenu'             ; 1
               dc i'ignore'             ; 2 in system window
               dc i'inContent'          ; 3 in content of window
               dc i'ignore'             ; 4 in drag
               dc i'ignore'             ; 5 in grow
               dc i'DoGoAway'           ; 6 in goaway
               dc i'ignore'             ; 7 in zoom
               dc i'ignore'             ; 8 in info bar
               dc i'ignore'             ; 9 in ver scroll
               dc i'ignore'             ; 10 in hor scroll
               dc i'ignore'             ; 11 in frame
               dc i'ignore'             ; in drop

               END
